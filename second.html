AND GATE



VHDL CODE: 
 library ieee;
 use ieee.std_logic_1164.all;

 entity andgate is
 port(
      a,b: in std_logic;
	  c: out std_logic
	  );
 end entity andgate;

 architecture behaviour of andgate is 
 begin
      c < = a and b;
 end architecture behaviour;
